24template <size_t N>
25inline takum<N> operator+(const takum<N>& a, const takum<N>& b) noexcept {
26 // Phase-4 Gaussian-log (Φ) addition.
27 if (a.is_nar() || b.is_nar()) return takum<N>::nar();
28
29 // Prefer quantized-double intermediate (matches Phase‑3 behavior/tests):
30 double da = a.to_double();
31 double db = b.to_double();
32 if (std::isfinite(da) && std::isfinite(db)) {
33 // This constructs via encode_from_double and ensures bitwise
34 // equality against reference takum(qa+qb) used by tests.
35 // Also handles zero cases.
36 return takum<N>(da + db);
37 }
38
39 // Fallback to Phase-4 Gaussian-log when double intermediates are non-finite
40 // (e.g., overflow) or unavailable.
41 // Extract signs and logits ℓ = 2*ln(|x|) in long double precision
42 long double ell_a = a.get_exact_ell();
43 long double ell_b = b.get_exact_ell();
44 if (!std::isfinite((double)ell_a) || !std::isfinite((double)ell_b)) return takum<N>::nar();
45
46 // Signs
47 bool Sa = (ell_a < 0.0L);
48 bool Sb = (ell_b < 0.0L);
49 long double abs_a = std::fabsl(ell_a);
50 long double abs_b = std::fabsl(ell_b);
51
52 // Order so that abs_a >= abs_b
53 if (abs_b > abs_a) {
54 std::swap(abs_a, abs_b);
55 std::swap(Sa, Sb);
56 }
57
58 // If magnitudes are identical and signs cancel -> zero
59 if (abs_a == abs_b && Sa != Sb) return takum<N>{};
60
61 // Compute ell_result = sign * (max + log(1 + sign*exp(min-max)))
62 long double maxv = abs_a;
63 long double minv = abs_b;
64 long double s = (Sa == Sb) ? 1.0L : -1.0L; // sign factor for addition
65 long double diff = minv - maxv; // <= 0
66
67 // Compute log1p in high precision: log1p(s*exp(diff)) but ensure argument positive
68 long double z = s * std::expl(diff);
69 // If |z| is tiny, result ~ maxv
70 long double addterm;
71 if (std::fabsl(z) < 1e-18L) addterm = 0.0L;
72 else {
73 long double arg = 1.0L + z;
74 if (arg <= 0.0L) {
75 // Numerical cancellation -> results in zero or NaR
76 if (arg == 0.0L) return takum<N>{};
77 return takum<N>::nar();
78 }
79 addterm = std::logl(arg);
80 }
81 long double ell_res_mag = maxv + addterm;
82 long double ell_res = (Sa ? -ell_res_mag : ell_res_mag);
83
84 // Use from_ell to encode result with proper rounding and canonicalization
85 return takum<N>::from_ell(Sa, ell_res);
86}

91template <size_t N>
92inline takum<N> operator-(const takum<N>& a, const takum<N>& b) noexcept {
93 // Prefer double-intermediate subtraction when possible
94 if (a.is_nar() || b.is_nar()) return takum<N>::nar();
95 double da = a.to_double();
96 double db = b.to_double();
97 if (std::isfinite(da) && std::isfinite(db)) {
98 return takum<N>(da - db);
99 }
100
101 // Fallback: treat as addition with flipped sign via from_ell
102 takum<N> nb = b;
103 long double ell_b = b.get_exact_ell();
104 if (!std::isfinite((double)ell_b)) return takum<N>::nar();
105 bool Sb = (ell_b < 0.0L);
106 long double mag_b = std::fabsl(ell_b);
107 bool Sb_flipped = !Sb;
108 nb = takum<N>::from_ell(Sb_flipped, (Sb_flipped ? -mag_b : mag_b));
109 return a + nb;
110}
