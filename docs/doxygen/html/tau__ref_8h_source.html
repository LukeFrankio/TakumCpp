37template <size_t N>
38inline uint64_t encode_double_to_bits(double x) noexcept {
39 static_assert(N >= 12 && N <= 64, "Takum reference encoder supports 12..64 bits");
40
41 const double sqrt_e = std::sqrt(std::exp(1.0));
42 const double ln2 = std::log(2.0);
43 const double max_exp = 255.0;
44
45 // NaR pattern: S=1, D=0, R=000 (r=7), C=0000000 (7 bits), M all 0 (p=N-12)
46 uint64_t nar_pattern = uint64_t(1) << (N - 1); // S=1 at MSB
47 if (std::isnan(x)) return nar_pattern;
48 if (x == 0.0) return 0u;
49 if (!std::isfinite(x)) return nar_pattern; // NaR for inf
50
51 bool s = std::signbit(x);
52 double abs_x = std::fabs(x);
53 if (abs_x < std::pow(sqrt_e, -max_exp)) return s ? nar_pattern : 0u; // underflow to 0 or -0, but since packed, 0
54 if (abs_x > std::pow(sqrt_e, max_exp)) return nar_pattern; // overflow to NaR
55
56 // Decompose x = (-1)^s * (1 + f) * 2^h (IEEE-like frexp for base 2)
57 int h_exp;
58 double f = std::frexp(abs_x, &h_exp); // f in [0.5, 1), h_exp such that abs_x = f * 2^h_exp
59 double significand = 2.0 * f; // now [1, 2)
60 double ln_sig = std::log(significand);
61 double ell = 2.0 * (h_exp * ln2 + ln_sig); // â„“ = 2 * ln(|x|)
62 int64_t c = static_cast<int64_t>(std::floor(ell));
63 double m = ell - static_cast<double>(c);
64
65 bool D = (c >= 0);
66 int64_t abs_c = std::abs(c);
67 int r;
68 if (D) {
69 r = (abs_c == 0) ? 0 : static_cast<int>(std::floor(std::log2(static_cast<double>(abs_c + 1))));
70 } else {
71 r = (abs_c == 0) ? 0 : static_cast<int>(std::floor(std::log2(static_cast<double>(abs_c))));
72 }
73 r = std::max(0, std::min(7, r)); // clamp to 0..7
74
75 uint64_t R_val = D ? static_cast<uint64_t>(r) : static_cast<uint64_t>(7 - r);
76 uint64_t R = (R_val & 4) >> 2 | (R_val & 2) >> 1 | (R_val & 1);
77
78 // Characteristic bits C (r bits)
79 uint64_t C_val;
80 if (D) {
81 C_val = static_cast<uint64_t>(c - (1LL << r) + 1);
82 } else {
83 C_val = static_cast<uint64_t>(c + (1LL << (r + 1)) - 1);
84 }
85 uint64_t C = 0;
86 for (int i = 0; i < r; ++i) {
87 C |= (C_val & 1u) << i;
88 C_val >>= 1;
89 }
90
91 // Mantissa bits M (p = N - r - 5 bits)
92 int p = N - r - 5;
93 if (p < 0) p = 0; // though for N>=12, r<=7, p>=0
94 uint64_t M_power = (p == 0) ? 1ULL : (1ULL << p);
95 uint64_t M_scaled = static_cast<uint64_t>(std::floor(m * static_cast<double>(M_power)));
96 uint64_t M_bits = M_scaled;
97
98 // Pack: S (1), D (1), R (3), C (r), M (p)
99 uint64_t bits = 0;
100 bits |= static_cast<uint64_t>(s) << (N - 1);
101 bits |= static_cast<uint64_t>(D) << (N - 2);
102 bits |= (R << (N - 5));
103 int char_start = N - 5 - r;
104 bits |= (C << char_start);
105 int mant_start = char_start - p;
106 bits |= (M_bits << mant_start);
107
108 return bits;
109}

126template <size_t N>
127inline double decode_bits_to_double(uint64_t bits) noexcept {
128 static_assert(N >= 12 && N <= 64, "Takum reference decoder supports 12..64 bits");
129
130 const double sqrt_e = std::sqrt(std::exp(1.0));
131
132 // NaR: S=1, D=0, R=000, C=0000000 (r=7), M=0...0 (p=N-12)
133 bool is_nar = (bits & (1ULL << (N-1))) != 0; // S=1
134 uint64_t low_bits = bits & ((1ULL << (N-1)) - 1);
135 if (is_nar && low_bits == 0) return std::numeric_limits<double>::quiet_NaN();
136 if (bits == 0) return 0.0;
137
138 bool S = (bits & (1ULL << (N-1))) != 0;
139 bool D = (bits & (1ULL << (N-2))) != 0;
140 uint64_t R = (bits >> (N-5)) & 7u;
141 int r = D ? static_cast<int>(R) : 7 - static_cast<int>(R);
142
143 int char_start = N - 5 - r;
144 uint64_t C = (bits >> char_start) & ((1ULL << r) - 1);
145
146 int p = N - r - 5;
147 int mant_start = char_start - p;
148 uint64_t M = (p > 0) ? (bits >> mant_start) & ((1ULL << p) - 1) : 0;
149
150 // Compute c
151 int64_t c_sum = 0;
152 for (int i = 0; i < r; ++i) {
153 c_sum += ((C >> i) & 1) * (1LL << i);
154 }
155 int64_t c;
156 if (D) {
157 c = (1LL << r) - 1 + c_sum;
158 } else {
159 c = - (1LL << (r + 1)) + 1 + c_sum;
160 }
161
162 // Compute m
163 double m = 0.0;
164 if (p > 0) {
165 double m_power = static_cast<double>(1ULL << p);
166 m = static_cast<double>(M) / m_power;
167 }
168
169 double ell = static_cast<double>(c) + m;
170 double y = std::pow(sqrt_e, ell);
171 return S ? -y : y;
172}
