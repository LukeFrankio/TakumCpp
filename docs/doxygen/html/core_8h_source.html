186 if constexpr (N > 64) {
187 // Multi-word signed compare: perform sign-aware lexicographic compare
188 // Words are stored little-endian (word 0 is least-significant). Determine
189 // the index and bit of the sign and perform comparison starting from
190 // the most-significant word.
191 size_t words = storage.size();
192 // compute msb word index
193 size_t msb_word = (N - 1) / 64;
194 // Sign bits assumed stored in msb_word at position msb_bit

196 uint64_t other_msw = other.storage[msb_word];
197 // Signed compare by lexicographic order with top word sign extension
198 for (size_t wi = msb_word + 1; wi-- > 0;) {
199 uint64_t a = storage[wi];
200 uint64_t b = other.storage[wi];
201 if (a == b) {
202 if (wi == 0) break;

205 // For the top word, perform signed comparison limited to used bits
206 if (wi == msb_word) {
207 int used_bits = ((N - 1) % 64) + 1;
208 uint64_t mask = (used_bits == 64) ? ~0ULL : ((1ULL << used_bits) - 1ULL);
209 uint64_t a_top = a & mask;

211 // sign-extend to 128-bit conceptual: compare as signed of used_bits
212 int64_t a_signed = static_cast<int64_t>( (a_top << (64 - used_bits)) ) >> (64 - used_bits);
213 int64_t b_signed = static_cast<int64_t>( (b_top << (64 - used_bits)) ) >> (64 - used_bits);
214 return a_signed < b_signed;

217 return a < b ? true : false;

222 // Treat the low N bits as a signed integer by shifting to MSB and arithmetic right shift.

446 // For very large N we still provide a double diagnostic only.
447 if (is_zero()) return 0.0;
448 // Use high-precision decode where available for N<=64

530 static storage_t encode_from_double(double x) noexcept {
531 if constexpr (N <= 64) {
532 // Small-width path: build uint64_t then widen/copy
533 uint64_t small = encode_from_double_u64(x);
534 return static_cast<storage_t>(small);
535 } else {
536 // Multi-word: write the bit fields across the words using helpers.
537 storage_t out{};
538 out.fill(0);
539 // If x is zero or non-finite the small path already encodes it in low bits
540 // but we must still place fields at correct offsets across words.
541 if (x == 0.0) return out;
542 if (!std::isfinite(x)) { out = takum::nar().storage; return out; }
543
544 bool S = std::signbit(x);
545 long double abs_x = std::fabsl(x);
546 long double ell = 2.0L * std::logl(abs_x);
547
548 long double clamp_pos = max_ell();
549 // Per spec: values outside representable dynamic range map to NaR
550 if (ell > clamp_pos || ell < -clamp_pos) {
551 out = takum::nar().storage;
552 return out;
553 }
554
555 int64_t c = static_cast<int64_t>(std::floorl(ell));
556 bool D = (c >= 0);
557 int64_t abs_c = D ? c : -c;
558 uint32_t r = (abs_c != 0)
559 ? static_cast<uint32_t>(std::floorl(std::log2l(D ? abs_c + 1 : abs_c)))
560 : 0;
561 r = std::min<uint32_t>(7, r);
562 uint32_t R = D ? r : (7U - r);
563
564 uint64_t c_bits = 0ULL;
565 if (r != 0) {
566 if (D) c_bits = static_cast<uint64_t>(c - ((1ULL << r) - 1ULL));
567 else c_bits = static_cast<uint64_t>(c + ((1ULL << (r+1)) - 1ULL));
568 }
569
570 long double m = ell - static_cast<long double>(c);
571 if (m < 0.0L) m = 0.0L;
572 if (m >= 1.0L) m = 0.999999L;
573
574 size_t p = N - 5 - static_cast<size_t>(r);
575
576 // Helper lambdas for multi-word bit writes
577 auto write_bit = [&](storage_t &dst, size_t bit_index, bool val) {
578 size_t wi = bit_index / 64;
579 size_t bi = bit_index % 64;
580 if constexpr (N <= 64) {
581 // not used here
582 } else {
583 if (val) dst[wi] |= (1ULL << bi);
584 else dst[wi] &= ~(1ULL << bi);
585 }
586 };

588 auto write_u64_at = [&](storage_t &dst, size_t bit_index, uint64_t value, size_t len) {
589 // Write up to 64 bits of `value` starting at bit_index (LSB-first), crossing words.
590 size_t remaining = len;
591 size_t pos = 0;
592 while (remaining > 0) {
593 size_t wi = bit_index / 64;
594 size_t bi = bit_index % 64;
595 size_t space = 64 - bi;
596 size_t take = std::min<size_t>(remaining, space);
597 uint64_t mask = (take == 64) ? ~0ULL : ((1ULL << take) - 1ULL);
598 uint64_t chunk = (value >> pos) & mask;
599 // clear destination bits then set
600 uint64_t dst_mask = ~(mask << bi);
601 dst[wi] &= dst_mask;
602 dst[wi] |= (chunk << bi);
603 bit_index += take;
604 pos += take;
605 remaining -= take;
606 }
607 };
608
609 // Place S at N-1
610 write_bit(out, N - 1, S);
611 // D at N-2
612 write_bit(out, N - 2, D);
613 // R: 3 bits at positions [N-5 .. N-3]
614 write_u64_at(out, N - 5, static_cast<uint64_t>(R), 3);
615 // C bits: r bits at offset p
616 if (r > 0) write_u64_at(out, p, c_bits, static_cast<size_t>(r));
617 // M bits: write p bits representing fractional part m; generate bits sequentially
618 long double frac = m;
619 for (size_t i = 0; i < p; ++i) {
620 frac *= 2.0L;
621 bool bit = (frac >= 1.0L);
622 if (bit) frac -= 1.0L;
623 write_bit(out, i, bit);
624 }
625
626 // Mask top bits just in case
627 mask_to_N(out);
628 return out;
629 }

671 uint64_t c_bits = 0ULL;

675 } else {

697 (static_cast<uint64_t>(D) << (N - 2)) | // Direction
698 (static_cast<uint64_t>(R) << (N - 5)); // Regime
699 packed |= (c_bits << p); // Characteristic
700 packed |= m_bits; // Mantissa

702 } else {

704 return 0ULL;

747 // Extract single-word bits for N<=64 to reuse previous logic
748 if constexpr (N <= 64) {

753 auto read_bit = [&](size_t bit_index) -> bool {

755 size_t bi = bit_index % 64;

759 auto read_u64_at = [&](size_t bit_index, size_t len) -> uint64_t {
760 // Read up to 64 bits starting at bit_index (LSB-first)
761 uint64_t acc = 0ULL;
762 size_t remaining = len;
763 size_t pos = 0;
764 while (remaining > 0) {
765 size_t wi = bit_index / 64;
766 size_t bi = bit_index % 64;
767 size_t space = 64 - bi;
768 size_t take = std::min<size_t>(remaining, space);
769 uint64_t mask = (take == 64) ? ~0ULL : ((1ULL << take) - 1ULL);
770 uint64_t chunk = (bits_storage[wi] >> bi) & mask;
771 acc |= (chunk << pos);
772 bit_index += take;
773 pos += take;
774 remaining -= take;
775 }

779 // Read sign and NaR/zero checks
780 bool S = read_bit(N - 1);
781 // Read lower (N-1) bits to check NaR or zero
782 bool lower_all_zero = true;
783 for (size_t i = 0; i < N - 1; ++i) if (read_bit(i)) { lower_all_zero = false; break; }
784 if (S && lower_all_zero) return std::numeric_limits<double>::quiet_NaN();
785 // If sign==0 and all lower bits zero, this is the zero pattern
786 if (!S && lower_all_zero) return 0.0;

788 bool D = read_bit(N - 2);

790 uint32_t r = D ? R : (7U - R);
791 size_t p = N - 5 - r;
792 uint64_t c_bits = (r == 0) ? 0ULL : read_u64_at(p, static_cast<size_t>(r));
793
794 int64_t c = 0;

796 else {
797 if (D) c = static_cast<int64_t>(((1ULL << r) - 1ULL) + c_bits);
798 else c = static_cast<int64_t>(-((1LL << (r + 1)) ) + 1LL + static_cast<int64_t>(c_bits));
799 }
800
801 // Read mantissa bits individually to build fractional value m
802 long double m = 0.0L;
803 long double twopow = 0.5L;
804 for (size_t i = 0; i < p; ++i) {
805 if (read_bit(i)) m += twopow;
806 twopow *= 0.5L;
807 }
808
809 long double ell = static_cast<long double>(c) + m;
810 double value_sign = S ? -1.0 : 1.0;
811 return value_sign * static_cast<double>(std::expl(ell * 0.5L));
812 }
813 }

825 static double decode_u64_to_double(uint64_t bits) noexcept {
826 if (bits == 0) return 0.0; // Zero per eq. (24)
827 // NaR check: S=1 and D=R=C=M=0 per Def. 2
828 bool S = (bits >> (N - 1)) & 1ULL; // Sign per eq. (14)
829 uint64_t lower = bits & ((N >= 64) ? ~0ULL : ((1ULL << (N - 1)) - 1ULL));
830 if (S && lower == 0) {
831 return std::numeric_limits<double>::quiet_NaN(); // NaR per eq. (24)
832 }
833 // Extract D per eq. (15)
834 bool D = (bits >> (N - 2)) & 1ULL;
835 // Extract R per eq. (16), compute r per eq. (17)
836 uint32_t R = ((bits >> (N - 5)) & 7ULL);
837 uint32_t r = D ? R : (7U - R);
838 // Extract C per eq. (18), compute c per eq. (19)
839 uint64_t c_mask = (r == 0) ? 0ULL : (((1ULL << r) - 1ULL) << (N - 5 - r));
840 uint64_t c_bits = (c_mask == 0ULL) ? 0ULL : ((bits & c_mask) >> (N - 5 - r));
841 int64_t c = D
842 ? (int64_t(((r == 0) ? 0ULL : ((1ULL << r) - 1ULL)) + c_bits)) // D=1 case eq. (19)
843 : (int64_t(-((1LL << (r + 1))) + 1LL + static_cast<int64_t>(c_bits))); // D=0 case eq. (19)
844 // p per eq. (20)
845 size_t p = N - 5 - r;
846 // Extract M per eq. (21), compute m per eq. (22)
847 uint64_t m_bits = (p == 0) ? 0ULL : (bits & ((1ULL << p) - 1ULL));
848 double m = (p > 0) ? (static_cast<double>(m_bits) * std::pow(2.0, -static_cast<int>(p))) : 0.0;
849
850 // â„“ = c + m per eq. (23)
851 double ell = static_cast<double>(c) + m;
852 // value = (-1)^S * exp(â„“ / 2) = (-1)^S * âˆše ^ â„“ per eq. (24)
853 double value_sign = S ? -1.0 : 1.0;
854 return value_sign * std::exp(ell * 0.5);
855 }
