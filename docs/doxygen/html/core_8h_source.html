181 bool operator<(const takum& other) const noexcept {
182 if (is_nar() || other.is_nar()) {
183 return is_nar() && !other.is_nar();
184 }
185 // Both real: signed integer compare with sign extension for N bits
186 if constexpr (N > 64) {
187 // Multi-word signed compare: perform sign-aware lexicographic compare
188 // Words are stored little-endian (word 0 is least-significant). Determine
189 // the index and bit of the sign and perform comparison starting from
190 // the most-significant word.
191 size_t words = storage.size();
192 // compute msb word index
193 size_t msb_word = (N - 1) / 64;
194 // Sign bits assumed stored in msb_word at position msb_bit
195 uint64_t self_msw = storage[msb_word];
196 uint64_t other_msw = other.storage[msb_word];
197 // Signed compare by lexicographic order with top word sign extension
198 for (size_t wi = msb_word + 1; wi-- > 0;) {
199 uint64_t a = storage[wi];
200 uint64_t b = other.storage[wi];
201 if (a == b) {
202 if (wi == 0) break;
203 continue;
204 }
205 // For the top word, perform signed comparison limited to used bits
206 if (wi == msb_word) {
207 int used_bits = ((N - 1) % 64) + 1;
208 uint64_t mask = (used_bits == 64) ? ~0ULL : ((1ULL << used_bits) - 1ULL);
209 uint64_t a_top = a & mask;
210 uint64_t b_top = b & mask;
211 // sign-extend to 128-bit conceptual: compare as signed of used_bits
212 int64_t a_signed = static_cast<int64_t>( (a_top << (64 - used_bits)) ) >> (64 - used_bits);
213 int64_t b_signed = static_cast<int64_t>( (b_top << (64 - used_bits)) ) >> (64 - used_bits);
214 return a_signed < b_signed;
215 }
216 // full-word unsigned comparison suffices for lower words
217 return a < b ? true : false;
218 }
219 return false; // equal
220 } else {
221 // Explicit sign extension for monotonic comparison (Gustafson criterion 6).
222 // Treat the low N bits as a signed integer by shifting to MSB and arithmetic right shift.
223 int64_t self_signed = static_cast<int64_t>(static_cast<uint64_t>(storage) << (64 - N)) >> (64 - N);
224 int64_t other_signed = static_cast<int64_t>(static_cast<uint64_t>(other.storage) << (64 - N)) >> (64 - N);
225 return self_signed < other_signed;
226 }
227 }

306 takum reciprocal() const noexcept {
307 if (is_nar()) return nar();
308
309 // Zero → NaR
310 if constexpr (std::is_integral_v<storage_t>) {
311 if (storage == 0) return nar();
312 } else {
313 bool all_zero = true;
314 for (auto w : storage) if (w != 0) { all_zero = false; break; }
315 if (all_zero) return nar();
316 }
317
318 takum res;
319
320 if constexpr (N <= 64) {
321 uint64_t bits = uint64_t(storage);
322
323 // Reciprocal core: ~bits + 1 (mod 2^N)
324 bits = (~bits + 1ULL) & ((N == 64) ? ~0ULL : ((1ULL << N) - 1ULL));
325
326 // *** Fix: flip sign bit back ***
327 bits ^= (1ULL << (N - 1));
328
329 res.storage = static_cast<storage_t>(bits);
330 } else {
331 // multiword version
332 for (size_t i = 0; i < res.storage.size(); ++i) res.storage[i] = ~storage[i];
333
334 // +1 with carry
335 bool carry = true;
336 for (size_t i = 0; i < res.storage.size() && carry; ++i) {
337 uint64_t prev = res.storage[i];
338 res.storage[i] = prev + 1ULL;
339 carry = (res.storage[i] == 0ULL);
340 }
341
342 // mask top word to N bits
343 size_t msb_word = (N - 1) / 64;
344 size_t used_bits = (N - 1) % 64 + 1;
345 uint64_t mask = (used_bits == 64) ? ~0ULL : ((1ULL << used_bits) - 1ULL);
346 res.storage[msb_word] &= mask;
347 for (size_t i = msb_word + 1; i < res.storage.size(); ++i) res.storage[i] = 0;
348
349 // *** Fix: flip sign bit back ***
350 res.storage[msb_word] ^= (1ULL << ((N - 1) % 64));
351 }
352
353 return res;
354 }

446 double get_exact_ell() const noexcept {
447 // Use decode helper that works for any N; return NaN for NaR
448 // For very large N we still provide a double diagnostic only.
449 if (is_zero()) return 0.0;
450 // Use high-precision decode where available for N<=64
451 if constexpr (N <= 64) {
452 uint64_t bits = uint64_t(storage);
453 // NaR check
454 bool S = (bits >> (N - 1)) & 1ULL;
455 uint64_t lower = bits & ((1ULL << (N - 1)) - 1ULL);
456 if (S && lower == 0) return std::numeric_limits<double>::quiet_NaN();
457 } else {
458 // Check NaR pattern via is_nar()
459 if (is_nar()) return std::numeric_limits<double>::quiet_NaN();
460 }
461 // Fallback: decode to double and then extract ell via log
462 double v = to_double();
463 if (!std::isfinite(v)) return std::numeric_limits<double>::quiet_NaN();
464 long double abs_v = std::fabsl(v);
465 long double ell = 2.0L * std::logl(abs_v);
466 return static_cast<double>((v < 0) ? -ell : ell);
467 }

532 static storage_t encode_from_double(double x) noexcept {
533 if constexpr (N <= 64) {
534 // Small-width path: build uint64_t then widen/copy
535 uint64_t small = encode_from_double_u64(x);
536 return static_cast<storage_t>(small);
537 } else {
538 // Multi-word: write the bit fields across the words using helpers.
539 storage_t out{};
540 out.fill(0);
541 // If x is zero or non-finite the small path already encodes it in low bits
542 // but we must still place fields at correct offsets across words.
543 if (x == 0.0) return out;
544 if (!std::isfinite(x)) { out = takum::nar().storage; return out; }
545
546 bool S = std::signbit(x);
547 long double abs_x = std::fabsl(x);
548 long double ell = 2.0L * std::logl(abs_x);
549
550 long double clamp_pos = max_ell();
551 // Per spec: values outside representable dynamic range map to NaR
552 if (ell > clamp_pos || ell < -clamp_pos) {
553 out = takum::nar().storage;
554 return out;
555 }
556
557 int64_t c = static_cast<int64_t>(std::floorl(ell));
558 bool D = (c >= 0);
559 int64_t abs_c = D ? c : -c;
560 uint32_t r = (abs_c != 0)
561 ? static_cast<uint32_t>(std::floorl(std::log2l(D ? abs_c + 1 : abs_c)))
562 : 0;
563 r = std::min<uint32_t>(7, r);
564 uint32_t R = D ? r : (7U - r);
565
566 uint64_t c_bits = 0ULL;
567 if (r != 0) {
568 if (D) c_bits = static_cast<uint64_t>(c - ((1ULL << r) - 1ULL));
569 else c_bits = static_cast<uint64_t>(c + ((1ULL << (r+1)) - 1ULL));
570 }
571
572 long double m = ell - static_cast<long double>(c);
573 if (m < 0.0L) m = 0.0L;
574 if (m >= 1.0L) m = 0.999999L;
575
576 size_t p = N - 5 - static_cast<size_t>(r);
577
578 // Helper lambdas for multi-word bit writes
579 auto write_bit = [&](storage_t &dst, size_t bit_index, bool val) {
580 size_t wi = bit_index / 64;
581 size_t bi = bit_index % 64;
582 if constexpr (N <= 64) {
583 // not used here
584 } else {
585 if (val) dst[wi] |= (1ULL << bi);
586 else dst[wi] &= ~(1ULL << bi);
587 }
588 };

590 auto write_u64_at = [&](storage_t &dst, size_t bit_index, uint64_t value, size_t len) {
591 // Write up to 64 bits of `value` starting at bit_index (LSB-first), crossing words.
592 size_t remaining = len;
593 size_t pos = 0;
594 while (remaining > 0) {
595 size_t wi = bit_index / 64;
596 size_t bi = bit_index % 64;
597 size_t space = 64 - bi;
598 size_t take = std::min<size_t>(remaining, space);
599 uint64_t mask = (take == 64) ? ~0ULL : ((1ULL << take) - 1ULL);
600 uint64_t chunk = (value >> pos) & mask;
601 // clear destination bits then set
602 uint64_t dst_mask = ~(mask << bi);
603 dst[wi] &= dst_mask;
604 dst[wi] |= (chunk << bi);
605 bit_index += take;
606 pos += take;
607 remaining -= take;
608 }
609 };
610
611 // Place S at N-1
612 write_bit(out, N - 1, S);
613 // D at N-2
614 write_bit(out, N - 2, D);
615 // R: 3 bits at positions [N-5 .. N-3]
616 write_u64_at(out, N - 5, static_cast<uint64_t>(R), 3);
617 // C bits: r bits at offset p
618 if (r > 0) write_u64_at(out, p, c_bits, static_cast<size_t>(r));
619 // M bits: write p bits representing fractional part m; generate bits sequentially
620 long double frac = m;
621 for (size_t i = 0; i < p; ++i) {
622 frac *= 2.0L;
623 bool bit = (frac >= 1.0L);
624 if (bit) frac -= 1.0L;
625 write_bit(out, i, bit);
626 }
627
628 // Mask top bits just in case
629 mask_to_N(out);
630 return out;
631 }

695 // Multi-word packing similar to encode_from_double multiword path
696 storage_t out{};
697 out.fill(0);

699 bool D;
700 int64_t c;
701 {
702 c = static_cast<int64_t>(std::floorl(ell_ld));
703 D = (c >= 0);

708 : 0;
709 r = std::min<uint32_t>(7, r);
710 uint32_t R = D ? r : (7U - r);

713 if (r != 0) {
714 if (D) c_bits = static_cast<uint64_t>(c - ((1ULL << r) - 1ULL));
715 else c_bits = static_cast<uint64_t>(c + ((1ULL << (r+1)) - 1ULL));
716 }

719 if (m < 0.0L) m = 0.0L;
720 if (m >= 1.0L) m = 0.999999L;
721
722 size_t p = N - 5 - static_cast<size_t>(r);

785 static uint64_t encode_from_double_u64(double x) noexcept {
786 if constexpr (N <= 64) {
787 if (x == 0.0) return 0ULL; // Zero representation per eq. (24)
788 if (!std::isfinite(x)) return static_cast<uint64_t>(takum::nar().storage); // NaN/Inf → NaR per NaR convention in Def. 2
789
790 bool S = std::signbit(x); // Sign bit per eq. (14)
791 long double abs_x = std::fabsl(x);
792
793 long double ell = 2.0L * std::logl(abs_x); // Logarithmic value ℓ = 2 * ln(|x|) for base √e, per eq. (23)
794
795 // Clamp |ℓ| to representable range |ℓ| < 255 per eq. (23)
796 long double clamp_pos = max_ell();
797 if (ell > clamp_pos) ell = clamp_pos;
798 if (ell < -clamp_pos) ell = -clamp_pos;
799
800 // Decompose ℓ into characteristic c = floor(ℓ) and mantissa m = ℓ - c per eq. (19) and (22)
801 int64_t c = static_cast<int64_t>(std::floorl(ell));
802 bool D = (c >= 0); // Direction bit: positive if c >= 0 per eq. (15)
803 int64_t abs_c = D ? c : -c;
804
805 // Regime r per eq. (17): floor(log2(|c| + D))
806 uint32_t r = (abs_c != 0)
807 ? static_cast<uint32_t>(std::floorl(std::log2l(D ? abs_c + 1 : abs_c)))
808 : 0;
809 r = std::min<uint32_t>(7, r); // Clamp to max regime 7
810 uint32_t R = D ? r : (7U - r); // Regime bits per eq. (16)
811
812 // Characteristic bits C per eq. (18), c per eq. (19)
813 uint64_t c_bits = 0ULL;
814 if (r != 0) {
815 if (D) {
816 c_bits = static_cast<uint64_t>(c - ((1ULL << r) - 1ULL)); // D=1 case
817 } else {
818 c_bits = static_cast<uint64_t>(c + ((1ULL << (r+1)) - 1ULL)); // D=0 case
819 }
820 }
821
822 // Mantissa m = fractional part of ℓ per eq. (22), p = N - 5 - r per eq. (20)
823 long double m = ell - static_cast<long double>(c);
824 if (m < 0.0L) m = 0.0L;
825 if (m >= 1.0L) m = 0.999999L; // avoid overflow in scaling
826
827 size_t p = N - 5 - static_cast<size_t>(r);
828 uint64_t m_bits = 0ULL;
829 if (p > 0 && m > 0.0L) {
830 long double m_power = std::ldexpl(1.0L, static_cast<int>(p));
831 long double m_scaled_ld = m * m_power;
832 m_bits = static_cast<uint64_t>(std::floorl(m_scaled_ld + 0.5L)); // Quantize m to p bits
833 uint64_t max_m = (p >= 64) ? ~0ULL : ((1ULL << p) - 1ULL);
834 if (m_bits > max_m) m_bits = max_m; // Clamp
835 }
836
837 // Pack bit fields into storage per Def. 2 bit layout: S D R C M
838 uint64_t packed = (static_cast<uint64_t>(S) << (N - 1)) | // Sign
839 (static_cast<uint64_t>(D) << (N - 2)) | // Direction
840 (static_cast<uint64_t>(R) << (N - 5)); // Regime
841 packed |= (c_bits << p); // Characteristic
842 packed |= m_bits; // Mantissa
843 return packed;
844 } else {
845 // For N>64 this helper is not used; provide a safe fallback
846 return 0ULL;
847 }
848 }
849
850private:
855 static constexpr uint64_t nbit_mask() noexcept {
856 if constexpr (N >= 64) return ~0ULL;
857 else return (N == 64 ? ~0ULL : ((1ULL << N) - 1ULL));
858 }
859
864 static void mask_to_N(storage_t& s) noexcept {
865 if constexpr (N <= 64) {
866 s = static_cast<storage_t>(uint64_t(s) & nbit_mask());
867 } else {
868 // Zero bits above N in the top word; lower words untouched.
869 size_t msb_word = (N - 1) / 64;
870 size_t used_bits_top = ((N - 1) % 64) + 1;
871 uint64_t top_mask = (used_bits_top == 64) ? ~0ULL : ((1ULL << used_bits_top) - 1ULL);
872 for (size_t i = msb_word + 1; i < s.size(); ++i) s[i] = 0; // safety
873 s[msb_word] &= top_mask;
874 }
875 }
876
888 static double decode_to_double(const storage_t& bits_storage) noexcept {
889 // Extract single-word bits for N<=64 to reuse previous logic
890 if constexpr (N <= 64) {
891 uint64_t bits = static_cast<uint64_t>(bits_storage);
892 return decode_u64_to_double(bits);
893 } else {
894 // Multi-word decode: read fields across words using bit-access helpers.
895 auto read_bit = [&](size_t bit_index) -> bool {
896 size_t wi = bit_index / 64;
897 size_t bi = bit_index % 64;
898 return ((bits_storage[wi] >> bi) & 1ULL) != 0ULL;
899 };
900
901 auto read_u64_at = [&](size_t bit_index, size_t len) -> uint64_t {
902 // Read up to 64 bits starting at bit_index (LSB-first)
903 uint64_t acc = 0ULL;
904 size_t remaining = len;
905 size_t pos = 0;
906 while (remaining > 0) {
907 size_t wi = bit_index / 64;
908 size_t bi = bit_index % 64;
909 size_t space = 64 - bi;
910 size_t take = std::min<size_t>(remaining, space);
911 uint64_t mask = (take == 64) ? ~0ULL : ((1ULL << take) - 1ULL);
912 uint64_t chunk = (bits_storage[wi] >> bi) & mask;
913 acc |= (chunk << pos);
914 bit_index += take;
915 pos += take;
916 remaining -= take;
917 }
918 return acc;
919 };
920
921 // Read sign and NaR/zero checks
922 bool S = read_bit(N - 1);
923 // Read lower (N-1) bits to check NaR or zero
924 bool lower_all_zero = true;
925 for (size_t i = 0; i < N - 1; ++i) if (read_bit(i)) { lower_all_zero = false; break; }
926 if (S && lower_all_zero) return std::numeric_limits<double>::quiet_NaN();
927 // If sign==0 and all lower bits zero, this is the zero pattern
928 if (!S && lower_all_zero) return 0.0;
929
930 bool D = read_bit(N - 2);
931 uint32_t R = static_cast<uint32_t>(read_u64_at(N - 5, 3));
932 uint32_t r = D ? R : (7U - R);
933 size_t p = N - 5 - r;
934 uint64_t c_bits = (r == 0) ? 0ULL : read_u64_at(p, static_cast<size_t>(r));
935
936 int64_t c = 0;
937 if (r == 0) c = 0;
938 else {
939 if (D) c = static_cast<int64_t>(((1ULL << r) - 1ULL) + c_bits);
940 else c = static_cast<int64_t>(-((1LL << (r + 1)) ) + 1LL + static_cast<int64_t>(c_bits));
941 }
942
943 // Read mantissa bits individually to build fractional value m
944 long double m = 0.0L;
945 long double twopow = 0.5L;
946 for (size_t i = 0; i < p; ++i) {
947 if (read_bit(i)) m += twopow;
948 twopow *= 0.5L;
949 }
950
951 long double ell = static_cast<long double>(c) + m;
952 double value_sign = S ? -1.0 : 1.0;
953 return value_sign * static_cast<double>(std::expl(ell * 0.5L));
954 }
955 }
956
967 static double decode_u64_to_double(uint64_t bits) noexcept {
968 if (bits == 0) return 0.0; // Zero per eq. (24)
969 // NaR check: S=1 and D=R=C=M=0 per Def. 2
970 bool S = (bits >> (N - 1)) & 1ULL; // Sign per eq. (14)
971 uint64_t lower = bits & ((N >= 64) ? ~0ULL : ((1ULL << (N - 1)) - 1ULL));
972 if (S && lower == 0) {
973 return std::numeric_limits<double>::quiet_NaN(); // NaR per eq. (24)
974 }
975 // Extract D per eq. (15)
976 bool D = (bits >> (N - 2)) & 1ULL;
977 // Extract R per eq. (16), compute r per eq. (17)
978 uint32_t R = ((bits >> (N - 5)) & 7ULL);
979 uint32_t r = D ? R : (7U - R);
980 // Extract C per eq. (18), compute c per eq. (19)
981 uint64_t c_mask = (r == 0) ? 0ULL : (((1ULL << r) - 1ULL) << (N - 5 - r));
982 uint64_t c_bits = (c_mask == 0ULL) ? 0ULL : ((bits & c_mask) >> (N - 5 - r));
983 int64_t c = D
984 ? (int64_t(((r == 0) ? 0ULL : ((1ULL << r) - 1ULL)) + c_bits)) // D=1 case eq. (19)
985 : (int64_t(-((1LL << (r + 1))) + 1LL + static_cast<int64_t>(c_bits))); // D=0 case eq. (19)
986 // p per eq. (20)
987 size_t p = N - 5 - r;
988 // Extract M per eq. (21), compute m per eq. (22)
989 uint64_t m_bits = (p == 0) ? 0ULL : (bits & ((1ULL << p) - 1ULL));
990 double m = (p > 0) ? (static_cast<double>(m_bits) * std::pow(2.0, -static_cast<int>(p))) : 0.0;
991
992 // ℓ = c + m per eq. (23)
993 double ell = static_cast<double>(c) + m;
994 // value = (-1)^S * exp(ℓ / 2) = (-1)^S * √e ^ ℓ per eq. (24)
995 double value_sign = S ? -1.0 : 1.0;
996 return value_sign * std::exp(ell * 0.5);
997 }
